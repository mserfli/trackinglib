45 value_type& dyNewEgo,

88auto EgoMotion<CovarianceMatrixPolicy_>::calcCircularMotionJacobian(const InertialMotion& motion,
89 value_type dt) -> math::SquareMatrix<value_type, 3, true>
90{
91 const auto T = dt;
92 const auto v = motion.v;
93 const auto a = motion.a;
94 const auto w = motion.w;
95
96 math::SquareMatrix<value_type, 3, true> J{};
97 J.setZeros();
98 J.at_unsafe(0, 0) = 2 * std::sin((0.5) * T * w) * std::cos((0.5) * T * w) / w;
99 J.at_unsafe(0, 1) = T * std::sin((0.5) * T * w) * std::cos((0.5) * T * w) / w;
100 J.at_unsafe(0, 2) = -0.5 * T * (T * a + 2 * v) * math::pow<2>(std::sin((0.5) * T * w)) / w +
101 (0.5) * T * (T * a + 2 * v) * math::pow<2>(std::cos((0.5) * T * w)) / w -
102 (T * a + 2 * v) * std::sin((0.5) * T * w) * std::cos((0.5) * T * w) / (w * w);
103 J.at_unsafe(1, 0) = 2 * math::pow<2>(std::sin((0.5) * T * w)) / w;
104 J.at_unsafe(1, 1) = T * math::pow<2>(std::sin((0.5) * T * w)) / w;
105 J.at_unsafe(1, 2) = T * (T * a + 2 * v) * std::sin((0.5) * T * w) * std::cos((0.5) * T * w) / w -
106 (T * a + 2 * v) * math::pow<2>(std::sin((0.5) * T * w)) / (w * w);
107 J.at_unsafe(2, 2) = T;

112template <typename CovarianceMatrixPolicy_>
113void EgoMotion<CovarianceMatrixPolicy_>::calcDisplacementVector(Displacement& displacement,
114 const InertialMotion& motion,
115 value_type dt)
116{
117 const value_type T = dt;
118 const value_type v = motion.v;
119 const value_type a = motion.a;
120 const value_type w = motion.w;
121 if (std::abs(T * w) < std::numeric_limits<value_type>::epsilon())
122 {
123 // special case for (T*w) very close to zero
124 displacement.vec.setZeros();
125 displacement.vec.at_unsafe(DS_X) = 0.5 * T * (2 * v + a * T);
126 displacement.sinDeltaPsi = static_cast<value_type>(0);
127 displacement.cosDeltaPsi = static_cast<value_type>(1);
128 }
129 else
130 {
131 // Calculate angular displacement: dφ = ω·T
132 const value_type dphi = w * dt;
133 const value_type dphi_2 = dphi / 2.0;
134 // Precompute trigonometric values for efficiency
135 const value_type sin_dphi = std::sin(dphi);
136 const value_type sin_dphi_2 = std::sin(dphi_2);
137 const value_type cos_dphi = std::cos(dphi);
138 const value_type cos_dphi_2 = std::cos(dphi_2);
139
140 // Calculate secant length
141 // c = (2·v·T + a·T^2)/dφ · sin(dφ/2) = T * (2·v + a·T)/dφ · sin(dφ/2)
142 const value_type c = T * (2.0 * v + a * T) / dphi * sin_dphi_2;
143
144 // Calculate displacement components
145 const value_type dx = c * cos_dphi_2;
146 const value_type dy = c * sin_dphi_2;
147
148 // Set displacement vector
149 displacement.vec.at_unsafe(DS_X) = dx;
150 displacement.vec.at_unsafe(DS_Y) = dy;
151 displacement.vec.at_unsafe(DS_PSI) = dphi;
152 displacement.sinDeltaPsi = sin_dphi;
153 displacement.cosDeltaPsi = cos_dphi;
154 }
155}

157template <typename CovarianceMatrixPolicy_>
158void EgoMotion<CovarianceMatrixPolicy_>::calcDisplacementCovariance(Displacement& displacement,
159 const math::SquareMatrix<value_type, 3, true>& J,
160 const InertialMotion& motion)
161{
162 // clang-format off
163 // Create diagonal matrix from motion uncertainties
164 auto diag =
165 math::DiagonalMatrix<value_type, 3>::FromList({
166 math::pow<2>(motion.sv), math::pow<2>(motion.sa), math::pow<2>(motion.sw)
167 });
168 // clang-format on
169
170 // Create covariance matrix using generic FromDiagonal constructor
171 displacement.cov = std::move(DisplacementCov::FromDiagonal(diag));
172
173 // Apply the transformation: P = J * Pin * J^T using apaT
174 displacement.cov.apaT(J);
175}
