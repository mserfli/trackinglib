104 void computeJacobian(JacobianMatrix& jacobian, const StateVec& state) const

109 const value_type range = std::sqrt(rangeSq);

111 jacobian.setZeros();
112 jacobian.at_unsafe(MEAS_RANGE, StateDef_::X) = x / range;
113 jacobian.at_unsafe(MEAS_RANGE, StateDef_::Y) = y / range;
114 jacobian.at_unsafe(MEAS_BEARING, StateDef_::X) = -y / rangeSq;
115 jacobian.at_unsafe(MEAS_BEARING, StateDef_::Y) = x / rangeSq;
116 if constexpr (motion::has_velocity_v<StateDef_>)
117 {
118 const value_type vx = state.at_unsafe(StateDef_::VX);

122 jacobian.at_unsafe(MEAS_DOPPLER, StateDef_::X) = (vx / range) - ((x * radial) / (rangeSq * range));
