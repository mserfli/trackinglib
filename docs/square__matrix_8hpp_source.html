60template <typename ValueType_, sint32 Size_, bool IsRowMajor_>
61inline auto SquareMatrix<ValueType_, Size_, IsRowMajor_>::determinant() const -> ValueType_
62{
63 // Create a copy of the matrix for LU decomposition
64 SquareMatrix<ValueType_, Size_, IsRowMajor_> luMatrix{*this};
65 sint32 permutationCount{0};
66
67 // Perform LU decomposition with partial pivoting
68 for (auto k = 0; k < Size_; ++k)
69 {
70 // Partial pivoting: find the row with maximum element in current column
71 auto maxRow = k;
72 ValueType_ maxVal = std::abs(luMatrix.at_unsafe(k, k));
73
74 for (auto i = k + 1; i < Size_; ++i)
75 {
76 const auto absVal = std::abs(luMatrix.at_unsafe(i, k));
77 if (absVal > maxVal)
78 {
79 maxVal = absVal;
80 maxRow = i;
81 }
82 }
83
84 // Swap rows if necessary
85 if (maxRow != k)
86 {
87 for (auto j = 0; j < Size_; ++j)
88 {
89 std::swap(luMatrix.at_unsafe(k, j), luMatrix.at_unsafe(maxRow, j));
90 }
91 permutationCount++;
92 }
93
94 // Check for singular matrix
95 if (std::abs(luMatrix.at_unsafe(k, k)) < std::numeric_limits<ValueType_>::epsilon())
96 {
97 return static_cast<ValueType_>(0);
98 }
99
100 // Perform elimination for rows below the current one
101 for (auto i = k + 1; i < Size_; ++i)
102 {
103 const auto factor = luMatrix.at_unsafe(i, k) / luMatrix.at_unsafe(k, k);
104
105 for (auto j = k; j < Size_; ++j)
106 {
107 luMatrix.at_unsafe(i, j) -= factor * luMatrix.at_unsafe(k, j);
108 }
109 }
110 }
111
112 // Calculate determinant as product of diagonal elements, multiplied by (-1)^permutationCount
113 ValueType_ det{1};
114 for (auto i = 0; i < Size_; ++i)
115 {
116 det *= luMatrix.at_unsafe(i, i);
117 }
118
119 // Apply sign based on number of row permutations
120 if (permutationCount % 2 != 0)
121 {
122 det = -det;
123 }
124
125 return det;
126}
