80template <typename ValueType_, sint32 Size_>
81inline auto CovarianceMatrixFactored<ValueType_, Size_>::at_unsafe(sint32 row, sint32 col) const -> ValueType_
82{
83 // cov(row,col) == cov(col,row), so we swap row and col if row > col
84 if (row > col)
85 {
86 std::swap(row, col);
87 }
88
89 ValueType_ result{};
90 // calc row, col element of _u * _d * _u.transpose()
91 // calc relevant elements of u*d to be rhs multiplied with uT(col::, col)==u(col, col::)
92 Vector<ValueType_, Size_> ud{};
93 for (auto i = col; i < Size_; ++i)
94 {
95 ud.at_unsafe(i) = _u.at_unsafe(row, i) * _d.at_unsafe(i);
96 }
97 MatrixColumnView<ValueType_, Size_, 1, true> udView{ud, 0, col, Size_ - 1};
98 MatrixRowView<ValueType_, Size_, Size_, true> uTView{_u, col, col, Size_ - 1};
99 result = uTView * udView; // calc the scalar product of ud*uT on relevant elements
100 return result;
101}

122template <typename ValueType_, sint32 Size_>
123inline auto CovarianceMatrixFactored<ValueType_, Size_>::inverse() const -> tl::expected<CovarianceMatrixFactored, Errors>
124{
125 // we use the transpose of the input matrix to get a column major matrix
126 const auto composed = tracking::math::SquareMatrix<ValueType_, Size_, false>{this->operator()().transpose()};
127 // we decompose the input matrix into LDLt form, with L being a column major lower triangular matrix
128 const auto ldlt = composed.decomposeLDLT();
129 if (ldlt.has_value())
130 {
131 const auto [l, d] = ldlt.value();
132 // we calc the inverse of L and D and map the inv(L).transpose() to U being again a row major upper triangular matrix
133 // the resulting UDUt matrix describes the covariance matrix in information form, i.e. the inverse covariance matrix
134 return CovarianceMatrixFactored{std::move(l.inverse().transpose()), std::move(d.inverse())};
135 }
136 else
137 {
138 return tl::unexpected<Errors>{ldlt.error()};
139 }
140}
