22template <typename ValueType_, sint32 Size_>
23inline auto CovarianceMatrixFull<ValueType_, Size_>::inverse() const -> tl::expected<CovarianceMatrixFull, Errors>
24{
25 const auto retVal = BaseSquareMatrix::decomposeLLT();
26 if (retVal.has_value())
27 {
28 const auto& L = *retVal;
29 // A * Ainv = eye(n,n) with A=L*L' from Cholesky decomposition
30 // L*(L'*Ainv) = eye(n,n)
31 // L*u = eye(n,n) -> solve for u using forward substitution on each column vector of eye(n,n)
32 const auto u = L.solve(CovarianceMatrixFull::Identity());
33 // L'*Ainv = u -> solve for Ainv using backward substitution
34 math::SquareMatrix<ValueType_, Size_, true> cov{L.transpose().solve(u)};
35 cov.symmetrize();
36 return CovarianceMatrixFull{std::move(cov)};
37 }
38 return tl::unexpected<Errors>{retVal.error()};
39}

42template <bool IsRowMajor_>
43inline void CovarianceMatrixFull<ValueType_, Size_>::apaT(const tracking::math::SquareMatrix<ValueType_, Size_, IsRowMajor_>& A)
44{
45 assert(this->isSymmetric() && "Covariance currently not symmetric");
46 // calculate only the upper triangle part of P and fill lower triangle part
47 BaseSquareMatrix cov{};
48 for (sint32 i = 0; i < Size_; ++i)
49 {
50 for (sint32 j = i; j < Size_; ++j)
51 {
52 ValueType_ element = static_cast<ValueType_>(0);
53 for (sint32 k = 0; k < Size_; ++k)
54 {
55 for (sint32 l = 0; l < Size_; ++l)
56 {
57 element += A.at_unsafe(i, k) * this->at_unsafe(k, l) * A.at_unsafe(j, l);
58 }
59 }
60 // construct symmetric covariance matrix by filling both upper and lower triangle
61 cov.at_unsafe(i, j) = element;
62 cov.at_unsafe(j, i) = element;
63 }
64 }
65 *this = CovarianceMatrixFull{std::move(cov)};
66}
