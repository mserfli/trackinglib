221 void convertStateVecIntoStateSpace()
222 {
223 const auto& y = this->getVec();
224 const auto& Y = this->getCovForInternalUse();
225 auto& x = this->getVecForInternalUse();
226 // precondition: y and Y must be in information space (Y = P^-1, y = Y * x)
227 // operation:
228 // Y * result = y
229 // result = Y^-1 * y = Y^-1 * Y * x = x
230 x = Y().qrSolve(y);
231 // postcondition: x is in state space, Y is unchanged (still in information space)
232 }
233
235 void convertStateVecIntoInformationSpace()
236 {
237 const auto& x = this->getVec();
238 const auto& Y = this->getCovForInternalUse();
239 auto& y = this->getVecForInternalUse();
240 // precondition: x must be in state space, Y must be in information space (Y = P^-1)
241 // operation:
242 // y = Y * x
243 y = static_cast<StateVec>(Y() * x);
244 // postcondition: y is in information space (y = Y * x), Y is unchanged (still in information space)
245 }
