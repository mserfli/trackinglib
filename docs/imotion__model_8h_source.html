172 void convertStateVecIntoStateSpace()
173 {
174 const auto& y = this->getVec();
175 const auto& Y = this->getCovForInternalUse();
176 auto& x = this->getVecForInternalUse();
177 // precondition: y and Y must be in information space (Y = P^-1, y = Y * x)
178 // operation:
179 // Y * result = y
