115{
116 using value_type = typename M::value_type;
117
118 // Access elements via at_unsafe() - works regardless of internal memory layout
119 // This is future-proof for packed triangular matrixes
120 for (sint32 row = 0; row < M::Rows; ++row)
121 {
122 for (sint32 col = 0; col < M::Cols; ++col)
123 {
124 if constexpr (std::is_floating_point_v<value_type>)
125 {
126 os << std::fixed << std::setprecision(10) << std::showpos << std::setw(16) << matrix.at_unsafe(row, col);

129 {
130 os << std::setw(8) << matrix.at_unsafe(row, col);
131 }

134 {
135 os << ", ";
136 }
