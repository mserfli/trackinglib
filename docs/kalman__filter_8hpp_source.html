39template <typename UpdateMode_, sint32 DimX_, sint32 DimZ_>
40inline void KalmanFilter<CovarianceMatrixPolicy_>::updateState(math::Vector<value_type, DimX_>& x,
41 CovarianceMatrixType<DimX_>& P,
42 const math::Vector<value_type, DimZ_>& innovation,
43 const math::Matrix<value_type, DimZ_, DimX_>& H,
44 const CovarianceMatrixType<DimZ_>& R)
45{
46 static_assert(std::is_same_v<UpdateMode_, update_mode::Block> || std::is_same_v<UpdateMode_, update_mode::Sequential>,
47 "UpdateMode_ must be update_mode::Block or update_mode::Sequential");
48 static_assert(!(CovarianceMatrixPolicy_::is_factored && std::is_same_v<UpdateMode_, update_mode::Block>),
49 "Block update would destroy the UDU factorization; use update_mode::Sequential for the factored policy");
50
51 if constexpr (std::is_same_v<UpdateMode_, update_mode::Block>)
52 {
53 // block EKF update with Joseph stabilized covariance:
54 // S = H*P*H' + R ; K = P*H'*inv(S) ; x = x + K*nu ; P = (I-K*H)*P*(I-K*H)' + K*R*K'
55 const math::Matrix<value_type, DimX_, DimZ_> PHt{P * H.transpose()};
56 math::SquareMatrix<value_type, DimZ_> S{(H * PHt) + R()};
57 S.symmetrize();
58 const auto Sinv = S.inverse();
59 const math::Matrix<value_type, DimX_, DimZ_> K{PHt * Sinv};
60
61 x += K * innovation;
62
63 const math::SquareMatrix<value_type, DimX_> IKH{math::SquareMatrix<value_type, DimX_>::Identity() - (K * H)};
64 P.apaT(IKH);
65 P += (K * R()) * K.transpose();
66 P.symmetrize();
67 }
68 else
69 {
70 // sequential scalar updates on the decorrelated measurement system
71 // nu' = inv(U)*nu ; H' = inv(U)*H ; R' = D with R = U*D*U'
72 // (no-op for an already diagonal R), then per row i:
73 // v = P*h_i' ; s = h_i*v + d_ii ; x += v*nu_i/s ; P -= v*v'/s
74 // the innovation of later rows is re-linearized around the entry state x0:
75 // nu_i_corr = nu_i - h_i*(x - x0)
76 math::Vector<value_type, DimZ_> nuEff{innovation};
77 math::Matrix<value_type, DimZ_, DimX_> Heff{H};
78 math::DiagonalMatrix<value_type, DimZ_> rDiag{};
79 if (!detail::decorrelateMeasurement<CovarianceMatrixPolicy_>(nuEff, Heff, rDiag, R))
80 {
81 return; // defensive: R could not be decomposed, skip the update
82 }
83
84 const math::Vector<value_type, DimX_> x0{x};
85 math::Vector<value_type, DimX_> hi{};
86 for (sint32 i = 0; i < DimZ_; ++i)
87 {
88 for (sint32 col = 0; col < DimX_; ++col)
89 {
90 hi.at_unsafe(col) = Heff.at_unsafe(i, col);
91 }
92
93 const auto Pfull{P()};
94 const math::Vector<value_type, DimX_> v{Pfull * hi};
95 const value_type s = (hi * v) + rDiag.at_unsafe(i);
96 if (!(s > static_cast<value_type>(0.0)))
97 {
98 continue; // defensive: skip measurement rows with a non-positive innovation variance
99 }
100
101 const math::Vector<value_type, DimX_> dx{x - x0};
102 const value_type nuCorr = nuEff.at_unsafe(i) - (hi * dx);
103 x += (nuCorr / s) * v;
104
105 // rank-1 downdate P -= v*v'/s via the shared covariance contract: Agee-Turner keeps the
106 // UDU factorization intact for the factored policy, the full policy subtracts the
107 // symmetric outer product directly
108 P.rank1Update(static_cast<value_type>(-1.0) / s, v);
109 }
110 }
