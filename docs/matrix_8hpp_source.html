47template <typename ValueType_, sint32 Rows_, sint32 Cols_, bool IsRowMajor_>
48inline auto Matrix<ValueType_, Rows_, Cols_, IsRowMajor_>::FromList(
49 const std::initializer_list<std::initializer_list<ValueType_>>& list) -> Matrix
50{
51 Matrix result{};
52
53 // Validate row count - input is always in logical row-major format
54 if (list.size() != static_cast<std::size_t>(Rows))
55 {
56 throw std::runtime_error("Matrix::FromList: expected " + std::to_string(Rows) + " rows, got " + std::to_string(list.size()));
57 }
58
59 // Validate column count for each row - input is always in logical row-major format
60 for (const auto& row : list)
61 {
62 if (row.size() != static_cast<std::size_t>(Cols))
63 {
64 throw std::runtime_error("Matrix::FromList: expected " + std::to_string(Cols) + " columns, got " +
65 std::to_string(row.size()));
66 }
67 }
68
69 if constexpr (IsRowMajor_)
70 {
71 // For row-major storage, copy data directly
72 auto iter = result.data().begin();
73 for (const auto& row : list)
74 {
75 std::copy(row.begin(), row.end(), iter);
76 iter += row.size();
77 }
78 }
79 else
80 {
81 // For column-major storage, transpose the input data
82 for (sint32 col = 0; col < Cols; ++col)
83 {
84 for (sint32 row = 0; row < Rows; ++row)
85 {
86 // Get element from row-major input
87 auto row_iter = list.begin();
88 std::advance(row_iter, row);
89 auto col_iter = row_iter->begin();
90 std::advance(col_iter, col);
91
92 // Store in column-major order
93 result.data()[col * Rows + row] = *col_iter;
94 }
95 }
96 }
97 return result;
98}

373 ResultMatrix result{ResultMatrix::Zeros()};
