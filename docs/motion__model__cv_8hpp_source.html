29 const value_type& vy = this->operator[](StateDefCV::VY);

31 const value_type sinDeltaPsiEgo = egoMotion.getDisplacementCog().sinDeltaPsi;
32 const value_type cosDeltaPsiEgo = egoMotion.getDisplacementCog().cosDeltaPsi;
33 const value_type deltaXEgo = egoMotion.getDisplacementCog().vec.at_unsafe(EgoMotionType::DS_X);
34 const value_type deltaYEgo = egoMotion.getDisplacementCog().vec.at_unsafe(EgoMotionType::DS_Y);
35 const value_type distCog2Ego = egoMotion.getGeometry().distCog2Ego;
36
37 Go.setZeros();
38 Go.at_unsafe(X, X) = cosDeltaPsiEgo;
39 Go.at_unsafe(X, Y) = sinDeltaPsiEgo;
40 Go.at_unsafe(Y, X) = -sinDeltaPsiEgo;
41 Go.at_unsafe(Y, Y) = cosDeltaPsiEgo;
42 Go.at_unsafe(VX, VX) = cosDeltaPsiEgo;
43 Go.at_unsafe(VX, VY) = sinDeltaPsiEgo;
44 Go.at_unsafe(VY, VX) = -sinDeltaPsiEgo;
45 Go.at_unsafe(VY, VY) = cosDeltaPsiEgo;
46
47 const value_type x0 = -deltaYEgo + y;
48 const value_type x1 = deltaXEgo - distCog2Ego - x;

51 Ge.at_unsafe(X, EgoMotionType::DS_Y) = -sinDeltaPsiEgo;

53 Ge.at_unsafe(Y, EgoMotionType::DS_X) = sinDeltaPsiEgo;
54 Ge.at_unsafe(Y, EgoMotionType::DS_Y) = cosDeltaPsiEgo;
55 Ge.at_unsafe(Y, EgoMotionType::DS_PSI) = -(x0 * sinDeltaPsiEgo) + (x1 * cosDeltaPsiEgo);
56 Ge.at_unsafe(VX, EgoMotionType::DS_PSI) = -(vx * sinDeltaPsiEgo) + (vy * cosDeltaPsiEgo);
57 Ge.at_unsafe(VY, EgoMotionType::DS_PSI) = -(vx * cosDeltaPsiEgo) - (vy * sinDeltaPsiEgo);
58}
