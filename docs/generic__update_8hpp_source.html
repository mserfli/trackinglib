138template <typename MotionModel_, typename CovarianceMatrixPolicy_>
139template <sint32 TotalDimZ_, sint32 DimX_, sint32 Offset_, typename First_, typename... Rest_>
140inline void Update<MotionModel_, CovarianceMatrixPolicy_>::stackObservations(
141 math::Vector<value_type, TotalDimZ_>& innovation,
142 math::Matrix<value_type, TotalDimZ_, DimX_>& H,
143 math::SquareMatrix<value_type, TotalDimZ_>& R,
144 const math::Vector<value_type, DimX_>& state,
145 const First_& first,
146 const Rest_&... rest)
147{
148 constexpr sint32 DimZ = First_::DimZ;
149 static_assert((Offset_ + DimZ) <= TotalDimZ_, "stacked observation dimensions exceed TotalDimZ");
150
151 // Jacobian block
152 typename First_::JacobianMatrix Hi{};
153 first.computeJacobian(Hi, state);
154 H.template setBlock<DimZ, DimX_, DimZ, DimX_, 0, 0, true, Offset_, 0>(Hi);
155
156 // innovation block (innovation handling, e.g. angle wrapping, is resolved on the concrete model)
157 const auto predicted = first.predictMeasurement(state);
158 const auto nu = first.computeInnovation(first.getVec(), predicted);
159 for (sint32 i = 0; i < DimZ; ++i)
160 {
161 innovation.at_unsafe(Offset_ + i) = nu.at_unsafe(i);
162 }
163
164 // measurement covariance block (composed to full form; models are assumed uncorrelated)
165 const auto& Ri = first.getCov()();
166 R.template setBlock<DimZ, DimZ, DimZ, DimZ, 0, 0, true, Offset_, Offset_>(Ri);
167
168 stackObservations<TotalDimZ_, DimX_, Offset_ + DimZ>(innovation, H, R, state, rest...);
169}
