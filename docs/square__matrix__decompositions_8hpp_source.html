19// Householder QR decomposition
20template <typename ValueType_, sint32 Size_, bool IsRowMajor_>
21inline auto SquareMatrix<ValueType_, Size_, IsRowMajor_>::householderQR() const
22 -> std::pair<SquareMatrix, TriangularMatrix<ValueType_, Size_, false, IsRowMajor_>>
23{
24 // implementation based on https://www.cs.cornell.edu/~bindel/class/cs6210-f09/lec18.pdf
25
26 // Initially, Q is an identity matrix because no orthogonal transformations have been applied yet.
27 SquareMatrix Q{SquareMatrix::Identity()};
28 // Initializes the upper triangular matrix R as a copy of the input matrix. This is the matrix
29 // that will be transformed to become upper triangular.
30 SquareMatrix R{*this};
31
32 // scale to reduce numerical issues
33 const auto [min, max] = R.minmax();
34 const auto scaleFactor = std::abs(min) > std::abs(max) ? min : max;
35 R /= scaleFactor;
36
37 using ColumnVector = Vector<ValueType_, Size_>;
38 static ColumnVector w{};
39 for (auto j = 0; j < Size_; ++j)
40 {
41 // Extract Size_-j rows of the j-th column as a Vector starting in row j.
42 w.setBlock(Size_ - j, 1, j, j, j, 0, R);
43 for (auto k = 0; k < j; ++k)
44 {
45 // set unused values to zero
46 w.at_unsafe(k) = static_cast<ValueType_>(0);
47 }
48
49 const ValueType_ normx = w.norm();
50 // Determines the sign of the j-th diagonal element of R.
51 const ValueType_ sign =
52 (R.at_unsafe(j, j) < static_cast<ValueType_>(0)) ? static_cast<ValueType_>(1) : static_cast<ValueType_>(-1);
53 const ValueType_ u1 = R.at_unsafe(j, j) - sign * normx;
54 const ValueType_ tau = -sign * u1 / normx; // Computes the parameter tau for the Householder transformation.
55
56 w /= u1; // Computes the Householder vector w.
57 w.at_unsafe(j) = static_cast<ValueType_>(1); // Sets the j-th row of w to 1 for convenience.
58 const auto wView = MatrixColumnView<ValueType_, Size_, 1, true>(w, 0, j); // create view starting in j-th row
59
60 // Update R using the Householder transformation
61 for (auto i = j; i < Size_; ++i) // cols
62 {
63 // R(j:end, i) = R(j:end, i) - tau * w * (w' * R(j:end, i));
64 const auto tau_dotRw = tau * (MatrixColumnView<ValueType_, Size_, Size_, IsRowMajor_>(R, i, j) * wView);
65 for (auto k = j; k < Size_; ++k) // rows
66 {
67 R.at_unsafe(k, i) -= tau_dotRw * wView.at_unsafe(k - j);
68 }
69 }
70
71 // Update Q using the Householder transformation
72 for (auto i = 0; i < Size_; ++i) // rows
73 {
74 // Q(i,j:end) = Q(i,j:end) - tau * (Q(i,j:end) * w) * w';
75 const auto tau_dotQw = tau * (MatrixRowView<ValueType_, Size_, Size_, IsRowMajor_>(Q, i, j) * wView);
76 for (auto k = j; k < Size_; ++k) // cols
77 {
78 Q.at_unsafe(i, k) -= tau_dotQw * wView.at_unsafe(k - j);
79 }
80 }
81 }
82 auto triuR = TriangularMatrix<ValueType_, Size_, false, IsRowMajor_>{std::move(R)};
83 triuR *= scaleFactor;
84 return std::make_pair(std::move(Q), std::move(triuR));
85}

126template <typename ValueType_, sint32 Size_, bool IsRowMajor_>
127inline auto SquareMatrix<ValueType_, Size_, IsRowMajor_>::decomposeLDLT() const
128 -> tl::expected<std::pair<TriangularMatrix<ValueType_, Size_, true, IsRowMajor_>, DiagonalMatrix<ValueType_, Size_>>, Errors>
129{
130 if (isSymmetric())
131 {
132 if (hasStrictlyPositiveDiagonalElems())
133 {
134 TriangularMatrix<ValueType_, Size_, true, IsRowMajor_> L{};
135 DiagonalMatrix<ValueType_, Size_> D{};
136 for (auto j = 0; j < Size_; ++j)
137 {
138 ValueType_ sum = this->at_unsafe(j, j);
139 for (auto k = 0; k < j; ++k)
140 {
141 sum -= D.at_unsafe(k) * L.at_unsafe(j, k) * L.at_unsafe(j, k);
142 }
143 D.at_unsafe(j) = sum;
144 L.at_unsafe(j, j) = static_cast<ValueType_>(1);
145
146 for (auto i = j + 1; i < Size_; ++i)
147 {
148 sum = this->at_unsafe(i, j);
149 for (auto k = 0; k < j; ++k)
150 {
151 sum -= D.at_unsafe(k) * L.at_unsafe(i, k) * L.at_unsafe(j, k);
152 }
153 L.at_unsafe(i, j) = sum / D.at_unsafe(j);
154 }
155 }
156 return std::make_pair(std::move(L), std::move(D));
157 }
158 return tl::unexpected<Errors>{Errors::matrix_not_positive_definite};
159 }
160 return tl::unexpected<Errors>{Errors::matrix_not_symmetric};

163// UDUT decomposition
164template <typename ValueType_, sint32 Size_, bool IsRowMajor_>
165inline auto SquareMatrix<ValueType_, Size_, IsRowMajor_>::decomposeUDUT() const
166 -> tl::expected<std::pair<TriangularMatrix<ValueType_, Size_, false, IsRowMajor_>, DiagonalMatrix<ValueType_, Size_>>, Errors>
167{
168 if (isSymmetric())
169 {
170 const auto& P = *this;
171 TriangularMatrix<ValueType_, Size_, false, IsRowMajor_> U{};
172 DiagonalMatrix<ValueType_, Size_> D{};
173 for (sint32 j = Size_ - 1; j >= 0; --j)
174 {
175 for (sint32 i = j; i >= 0; --i)
176 {
177 auto sigma = P.at_unsafe(i, j);
178 for (sint32 k = j + 1; k < Size_; ++k)
179 {
180 sigma -= U.at_unsafe(i, k) * D.at_unsafe(k) * U.at_unsafe(j, k);
181 }
182 if (i == j)
183 {
184 D.at_unsafe(j) = std::max(sigma, std::numeric_limits<ValueType_>::epsilon());
185 U.at_unsafe(j, j) = static_cast<ValueType_>(1.0);
186 }
187 else
188 {
189 U.at_unsafe(i, j) = sigma / D.at_unsafe(j);
190 }
191 }
192 }
193 return std::make_pair(std::move(U), std::move(D));
194 }
195 return tl::unexpected<Errors>{Errors::matrix_not_symmetric};
196}
